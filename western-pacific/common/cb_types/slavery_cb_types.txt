	war_name = WAR_NAME_SLAVE

	truce_days = 3650

	is_permanent = yes

	can_ask_to_join_war = no

	defender_can_call_allies = yes
	can_call_vassals = yes

	max_attacker_battle_score = 100
	max_defender_battle_score = 100
	infamy_modifier = 0.35

	allow_distant = yes

	can_use_gui = {
		ROOT = {
			mercenary = no
			any_realm_province = {
				OR = {
					any_neighbor_province = {
						owner = {
							OR = {
								character = FROM
								any_liege = {
									character = FROM
								}
							}
						}
					}
					any_neighbor_province = {
						is_land = no
						any_neighbor_province = {
							owner = {
								OR = {
									character = FROM
									any_liege = {
										character = FROM
									}
								}
							}
						}
					}
					any_neighbor_province = {
						is_land = no
						any_neighbor_province = {
							is_land = no
							any_neighbor_province = {
								owner = {
									OR = {
										character = FROM
										any_liege = {
											character = FROM
										}
									}
								}
							}
						}
					}
				}
			}
		}
	}

	can_use = {
		FROM = {
			independent = yes
			mercenary = no
		}
		ROOT = {
			mercenary = no
			is_ruler = yes
			higher_real_tier_than = BARON
			crownlaw_title = {
				holder_scope = {
					OR = {
						character = ROOT
						is_liege_of = ROOT
						liege = {
							is_liege_of = ROOT
						}
					}
				}
				OR = {
					trait = trait_slaver
					has_law = slavery_2
					has_law = slavery_3
				}
			}
		}
	}

	is_valid = {
		ROOT = {
			mercenary = no
			is_ruler = yes
			higher_real_tier_than = BARON
			crownlaw_title = {
				holder_scope = {
					OR = {
						character = ROOT
						is_liege_of = ROOT
						liege = {
							is_liege_of = ROOT
						}
					}
				}
				OR = {
					trait = trait_slaver
					has_law = slavery_2
					has_law = slavery_3
				}
			}
		}
		FROM = {
			mercenary = no
			is_ruler = yes
			independent = yes
		}
	}

	on_success = {
		FROM = {
			save_event_target_as = slave_target
			prestige = -100
			transfer_scaled_wealth = {
				to = ROOT
				value = 1
				min = 100
			}
		}
		ROOT = {
			IF = {
				limit = {
					NOT = {
						trait = trait_slaver
					}
				}
				add_trait = trait_slaver
			}
			character_event = {
				id = slavery_mt_events.41
			}
		}
	}

	on_fail = {
		ROOT = {
			prestige = -100
		}
	}

	on_reverse_demand = {
		ROOT = {
			prestige = -100
			transfer_scaled_wealth = {
				to = FROM
				value = 2
				min = 100
			}
			any_courtier = {
				limit = {
					culture = FROM
					OR = {
						prisoner = yes
						trait = trait_slave
					}
				}
				IF = {
					limit = {
						trait = trait_slave
					}
					remove_opinion = {
						modifier = opinion_master
						who = ROOT
					}
					reverse_remove_opinion = {
						modifier = opinion_slave
						who = ROOT
					}
					remove_trait = trait_slave
					move_character = FROM
				}
				prisoner = no
			}
		}
		FROM = {
			prestige = 100
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}

	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}

	defender_ai_defeat_worth = {
		factor = 100
	}
	ai_will_do = { 
		factor = 0.2 
		modifier = {
			factor = 5 # gotta grab up some slaves
			ROOT = { trait = cruel }
		}
		modifier = {
			factor = 5 # gotta grab up some slaves
			ROOT = { trait = trait_slaver }
		}
		modifier = {
			factor = 0 # cowards and cripples need not apply
			ROOT = { 
				OR = { 
					trait = craven 
					trait = incapable
					trait = inbred
					trait = infirm
				}
			}
		}	
		modifier = {
			factor = 1.5 # prefer to raid infidels
			FROM = { NOT = { religion = ROOT } }
		}
		modifier = {
			factor = 1.5 # prefer to raid foreigners
			FROM = { NOT = { culture = ROOT } }
		}	
		modifier = {
			factor = 0.5 # prefer to raid infidels
			FROM = { religion = ROOT }
		}
		modifier = {
			factor = 0.5 # prefer to raid foreigners
			FROM = { culture = ROOT } 
		}
		modifier = {
			factor = 2
			ROOT = { 
				OR = { 
					trait = proud
					trait = deceitful
				}
			}
		}		
	}

	}

			mercenary = no

				OR = {
					has_law = slavery_1
					has_law = slavery_2
					has_law = slavery_3
				}
				OR = {
					NOT = {
						has_title_flag = lost_anti_slave_war
					}
					had_title_flag = {
						flag = lost_anti_slave_war
						years = 20
					}

			}

	}

	is_valid = {
		ROOT = {

					has_law = slavery_1

					has_law = slavery_3
				}
			}
		}
	}

	on_success = {
		ROOT = {
			prestige = 100
		if = { 
			limit = { NOT = { trait = liberator }  }
			character_event = {
				id = Ac.7780
			}
		}
	}
		FROM = {
			crownlaw_title = {
				IF = {
					limit = {
						has_law = slavery_1
					}
					revoke_law = slavery_1
					add_law = slavery_0
					PREV = {
						any_realm_lord = {
							character_event = {
								id = slavery_mt_events.3
							}
						}
						any_realm_province = {
							limit = {
								OR = {
									has_province_modifier = slave_host_1
									has_province_modifier = slave_host_2
									has_province_modifier = slave_host_3
								}
							}
							remove_province_modifier = slave_host_1
							remove_province_modifier = slave_host_2
							remove_province_modifier = slave_host_3
						}
					}
				}
				IF = {
					limit = {
						has_law = slavery_2
					}
					revoke_law = slavery_2
					add_law = slavery_1
					PREV = {
						any_realm_province = {
							limit = {
								OR = {
									has_province_modifier = slave_host_1
									has_province_modifier = slave_host_2
									has_province_modifier = slave_host_3
								}
							}
							remove_province_modifier = slave_host_1
							remove_province_modifier = slave_host_2
							remove_province_modifier = slave_host_3
						}
					}
				}
				IF = {
					limit = {
						has_law = slavery_3
					}
					revoke_law = slavery_3
					add_law = slavery_2
				}
				clr_title_flag = lost_anti_slave_war
				set_title_flag = lost_anti_slave_war
			}
			prestige = -100
			transfer_scaled_wealth = {
				to = ROOT
				value = 1
				min = 100
			}
		}
	}

	on_fail = {
		ROOT = {
			prestige = -100
		}
	}

	on_reverse_demand = {
		ROOT = {
			prestige = -100
			transfer_scaled_wealth = {
				to = FROM
				value = 2
				min = 100
			}
		}
		FROM = {
			prestige = 100
		}
